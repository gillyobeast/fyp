\begin{defn}
A \emph{C$^\ast$-algebra} is a Banach algebra $(A, \| \cdot \|)$ with involution $\ast : A \to A$ with the condition that
\begin{align*}
	\|x ^\ast x\| = \|x\|^2 \mbox{ for all } x \in A.
\end{align*}
\end{defn}
